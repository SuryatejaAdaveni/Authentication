npm install cors              # Enables Cross-Origin Resource Sharing (Frontend ↔ Backend)
npm install express           # Web framework for Node.js
npm install mongoose          # ODM for MongoDB (Object Data Mapping)
npm install bcrypt            # For password hashing (not bcrytp — that’s a typo)
npm install dotenv            # Loads environment variables from .env file
npm install cookie-parser     # Parses cookies from HTTP requests
npm install twilio            # For SMS and phone call APIs
npm install nodemailer        # For sending emails via SMTP
